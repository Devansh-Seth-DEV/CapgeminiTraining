	2. Main-Class: com.example.MainClass

jar cfm MyApp.jar manifest.txt -C bin/ . # cf:= create file, m:= manifest, C:= where is my .classes file in and include it in that directory(bin/ .)

jar cfe MyApp.jar com.example.MainClass -C bin/ . # telling where the entry point is

java -jar MyApp.jar # Create the jar file

java -cp MyApp.jar com.example.MainClass # -cp:= class path (in MyApp.jar file) and run this MainClass
